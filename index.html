-- BindableEvent
local be = Instance.new("BindableEvent")
be.Event:Connect(function(msg)
  print("Received: "..msg)
end)
be:Fire("Hello Bindable!")

-- BindableFunction
local bf = Instance.new("BindableFunction")
bf.OnInvoke = function(num)
  return num * 2
end
print("Result: "..bf:Invoke(5)) -- prints 10
